inherit namespace with module:CreateJS

var myObject = Crisp.utilCreate({
    ns: 'util.path'
}).objIni().objData({
    a: {
        b: 'B',
        c: 'C'
    }
});

myObject.pathFind({
    path: 'a.*:',
    success: function( e ) {
        console.log('success:', e );
    },
    complete: function( e ) {
        console.log('complete:', e );
    }
});
// logs:
// success: B
// success: C
// complete: [ { data: 'B' }, { data: 'C' } ]
